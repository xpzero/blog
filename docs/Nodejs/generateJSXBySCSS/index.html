根据scss层级结构生成jsx

最近画样式比较多,每次都需要节点层级排布、样式嵌套的写,太费劲了,撸一套简单的scss内容生成jsx的代码,方便之后的代码开发

scss 代码​

.parent-class {
  &-first {
    &-child {
      &:hover {
      }
    }
  }

  &-second {
    &-child {
    }
  }
}

主要代码​

const fs = require("fs");
const readline = require("readline");
let rl = readline.createInterface({
  input: fs.createReadStream("./test.scss"),
});
let prefixList = [];
let first = "";
let res = [];
rl.on("line", (line) => {
  const str = line.trim();
  if (!str) {
    return;
  }
  let div = "";
  if (str.includes("}")) {
    prefixList.pop();
    if (res[res.length - 1]) {
      res.push("</div>");
    } else {
      res.pop();
    }
  } else if (str.includes("{")) {
    if (!res.length) {
      first = str.slice(1, str.length - 1).trim();
      res.push("<div className={`${ns}`}>");
    } else if (str.includes("&")) {
      if (str.includes(":")) {
        res.push(null);
      } else {
        res.push(
          `<div className={\`\${ns}-${str.slice(2, str.length - 1).trim()}\`}>`,
        );
      }
    }
  }
}).on("close", () => {
  res.unshift(`const ns = '${first}';`);

  fs.writeFileSync("./test2.html", res.join(""));
  console.log("--------------Generate jsx success! ☕️☕️☕️--------------");
  process.exit(0);
});
